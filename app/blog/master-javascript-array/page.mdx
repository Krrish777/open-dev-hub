# Importance Array Methods in JavaScript

Arrays in JS are Dynamic, i.e., They can be changed during the time of execution. They can store more than one Data Type (Unlike Java and C++ where we have to declare an array beforehand and specify the size and the data type like strings or integers). To make changes in the array we have some methods provided by JavaScript which are quite useful when it comes to performing operations on the array. Let's Study these methods.

1. #### Array.push()

- Adds an Element to the end of the array

```JavaScript
let arr = [1, 2, 3]
arr.push(4)
console.log(arr)
// [1, 2, 3, 4]
```

- Returns the new length of the array

```JavaScript
let arr = [1, 2, 3]
let length = arr.push(4);
console.log(length)
  // 4
```

- Array.push() can be used to push multiple elements into the array

```JavaScript
let arr = ['element1', 'element2', 'element3']
let returnVal = arr.push('element4', 'element5', 'element6')
console.log(arr, returnVal)
// [ 'element1', 'element2', 'element3', 'element4', 'element5', 'element6' ] 6
```


2. #### Array.unshift()


- Adds an Element to the beginning of the array

```JavaScript
let arr = [1, 2, 3]
arr.unshift(0)
console.log(arr)
// [0, 1, 2, 3]
```

- Returns the new length of the array

```JavaScript
let arr = [1, 2, 3]
let length = arr.unshift(0);
console.log(length)
  // 4
```

- Array.unshift() can be used to push multiple elements into the array

```JavaScript
let arr = ['element1', 'element2', 'element3']
let returnVal = arr.unshift('element4', 'element5', 'element6')
console.log(arr, returnVal)
// [ 'element4', 'element5', 'element6', 'element1', 'element2', 'element3' ] 6
```


3. #### Array.pop()


- Removes the last element from the array

```JavaScript
let arr = [1, 2, 3]
arr.pop()
console.log(arr)
// [1, 2]
```

- Returns the removed element

```JavaScript
let arr = [1, 2, 3]
let removedElement = arr.pop();
console.log(removedElement)
  // 3
```


4. #### Array.shift()


- Removes the first element from the array

```JavaScript
let arr = [1, 2, 3]
arr.shift()
console.log(arr)
// [2, 3]
```

- Returns the removed element

```JavaScript
let arr = [1, 2, 3]
let removedElement = arr.shift();
console.log(removedElement)
  // 1
```


5. #### Array.splice()

- Adds/Removes elements from the array

```JavaScript
let arr = [1, 2, 3]
arr.splice(1, 0, 4)
console.log(arr)
// [1, 4, 2, 3]
```

- Returns the removed elements

```JavaScript
let arr = [1, 2, 3]
let removedElement = arr.splice(1, 1);
console.log(removedElement)
  // [2]
```

- Array.splice() can be used to add multiple elements into the array

```JavaScript
let arr = ['element1', 'element2', 'element3']
let returnVal = arr.splice(1, 0, 'element4', 'element5', 'element6')
console.log(arr, returnVal)
// [ 'element1', 'element4', 'element5', 'element6', 'element2', 'element3' ] []
```


6. #### Array.slice()


- Returns a shallow copy of the array

```JavaScript
let arr = [1, 2, 3]
let copyArr = arr.slice()
console.log(copyArr)
// [1, 2, 3]
```

- Array.slice() can be used to copy a part of the array

```JavaScript
let arr = [1, 2, 3, 4, 5]
let copyArr = arr.slice(1, 4)
console.log(copyArr)
// [2, 3, 4]
```


7. #### Array.concat()


- Merges two or more arrays

```JavaScript
let arr1 = [1, 2, 3]
let arr2 = [4, 5, 6]
let mergedArr = arr1.concat(arr2)
console.log(mergedArr)
// [1, 2, 3, 4, 5, 6]
```

- Array.concat() can be used to merge multiple arrays

```JavaScript
let arr1 = [1, 2, 3]
let arr2 = [4, 5, 6]
let arr3 = [7, 8, 9]
let mergedArr = arr1.concat(arr2, arr3)
console.log(mergedArr)
// [1, 2, 3, 4, 5, 6, 7, 8, 9]
```
